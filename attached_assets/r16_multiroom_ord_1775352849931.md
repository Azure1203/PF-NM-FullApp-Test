## FIX 1 — Multi-Room ORD Format

### How it works (from the reference file)

A multi-room Cabinet Vision ORD file has this structure:

```
[Header]
Version=4
Unit=1
Name="ORDER_NAME"
Description="ORDER_NAME"
PurchaseOrder="PO_NUMBER"
Comment=""
Customer="Perfect Fit Closets"
Address1="100-111 5 Avenue Southwest"

[Walls]

[Catalog]
Name="Perfect Fit"
Materials="..."
CabinetConstruction="X-Perfect Fit"
DrawerBoxConstruction="X-Perfect Fit Drawer 19"
RollOutConstruction="X-Perfect Fit Roll Out 19"

[Parameters]
Attribute="Banding","xPFC_BAND","text","EENN"

[Cabinets]
1,"34SHFF",355.6,752.5,19.0,"*","N",6,"",,0.0,0.0,0.0,1,0,"","","S"
;
```

**Key differences from standard format:**

1. **One `[Header]` for the entire file** — `Name` = project name (e.g. "H Holtermann"), `PurchaseOrder` = order ID
2. **Empty `[Walls]` section** — this MUST be present (even if empty) to enable the extended cabinet format with room numbers
3. **Extended `[Cabinets]` entries** — 18 fields instead of 8:

```
ORDER_NUM,"SKU",WIDTH,HEIGHT,DEPTH,"HINGE","END_TYPE",QTY,"COMMENT",,X,Y,Z,ROOM_NUM,0,"","","S"
```

Fields:
- 1: Sequential order entry number (global across all rooms)
- 2: Cabinet nomenclature (SKU)
- 3-5: Width, Height, Depth
- 6: Hinge ("*")
- 7: End type ("N")
- 8: Quantity
- 9: Comment (empty "")
- 10: (empty)
- 11-13: Position X, Y, Z (all 0.0)
- **14: Room number** — THIS is how rooms are defined. Items with `1` go to room 1, items with `2` go to room 2, etc.
- 15: 0
- 16: (empty "")
- 17: (empty "")
- 18: Fixture type ("S" = Standard)

4. **`[Parameters]` uses `Note=` for banding** instead of `Attribute=` for the banding line:
```
Note="Banding","xPFC_BAND","text","EENN"
```

### Implementation

**Each CSV file uploaded = one room.** Room number is assigned by file order (first file = room 1, second file = room 2, etc.).

The ORD `Name` field = project name (e.g. "H Holtermann"). Each room corresponds to a CSV file (e.g. "Her Holtermann V2", "His Walk In V3").

### Update the ORD download endpoint

Replace `GET /api/orders/:id/download/ord`:

```ts
app.get('/api/orders/:id/download/ord', isAuthenticated, async (req, res) => {
  try {
    const projectId = parseInt(req.params.id);
    const project = await storage.getProject(projectId);
    if (!project) return res.status(404).json({ message: 'Project not found' });
    
    const files = await storage.getProjectFiles(projectId);
    const allItems = await storage.getOrderItemsByProject(projectId);
    
    // Build the [Header] section
    const projectName = project.name || `Order ${projectId}`;
    const poNumber = project.orderId || '';
    
    let ord = '';
    ord += `[Header]\n`;
    ord += `Version=4\n`;
    ord += `Unit=1\n`;
    ord += `Name="${projectName}"\n`;
    ord += `Description="${projectName}"\n`;
    ord += `PurchaseOrder="${poNumber}"\n`;
    ord += `Comment=""\n`;
    ord += `Customer="Perfect Fit Closets"\n`;
    ord += `Address1="100-111 5 Avenue Southwest"\n`;
    ord += `\n`;
    ord += `[Walls]\n`;
    ord += `\n`;
    
    // Global sequential order entry number across all rooms
    let orderEntryNum = 1;
    
    for (let fileIdx = 0; fileIdx < files.length; fileIdx++) {
      const file = files[fileIdx];
      const roomNum = fileIdx + 1; // Room 1, 2, 3, ...
      
      // Get ORD-type items for this file only
      const fileItems = allItems.filter(i => 
        i.fileId === file.id && 
        i.exportType === 'ORD' && 
        i.exportText
      );
      
      for (const item of fileItems) {
        // Parse the existing exportText to extract [Catalog], [Parameters], and [Cabinets] data
        // The exportText currently contains the full block like:
        // [Catalog]
        // Name="Perfect Fit"
        // ...
        // [Parameters]
        // Attribute="Banding","xPFC_BAND","text","EENN"
        // [Cabinets]
        // 1,"34SHFF",355.6,752.5,19,"*","N",6
        // ;
        
        const exportBlock = item.exportText || '';
        
        // Extract the [Catalog] section (everything between [Catalog] and [Parameters])
        const catalogMatch = exportBlock.match(/\[Catalog\]([\s\S]*?)(?=\[Parameters\])/);
        const catalogSection = catalogMatch ? catalogMatch[1].trim() : '';
        
        // Extract the banding value from [Parameters]
        const bandingMatch = exportBlock.match(/Attribute="Banding","xPFC_BAND","text","([^"]+)"/);
        const bandingValue = bandingMatch ? bandingMatch[1] : 'NNNN';
        
        // Extract the cabinet line from [Cabinets]
        const cabinetMatch = exportBlock.match(/\[Cabinets\]\s*\n\s*(\d+,"[^"]+".+)/);
        const cabinetLine = cabinetMatch ? cabinetMatch[1] : null;
        
        if (!cabinetLine) continue;
        
        // Parse the existing cabinet line fields (standard 8-field format)
        // Format: 1,"SKU",W,H,D,"*","N",QTY
        const parts = cabinetLine.match(/(\d+),"([^"]+)",([^,]+),([^,]+),([^,]+),"([^"]+)","([^"]+)",(\d+)/);
        if (!parts) continue;
        
        const sku = parts[2];
        const width = parts[3];
        const height = parts[4];
        const depth = parts[5];
        const hinge = parts[6];
        const endType = parts[7];
        const qty = parts[8];
        
        // Write [Catalog] section
        ord += `[Catalog]\n`;
        ord += catalogSection + '\n';
        ord += `\n`;
        
        // Write [Parameters] section with Note= for banding (extended format)
        ord += `[Parameters]\n`;
        ord += `Note="Banding","xPFC_BAND","text","${bandingValue}"\n`;
        ord += `\n`;
        
        // Write [Cabinets] section with extended 18-field format including room number
        ord += `[Cabinets]\n`;
        ord += `${orderEntryNum},"${sku}",${width},${height},${depth},"${hinge}","${endType}",${qty},"",,0.0,0.0,0.0,${roomNum},0,"","","S"\n`;
        ord += `\n`;
        
        orderEntryNum++;
      }
    }
    
    if (orderEntryNum === 1) {
      return res.status(404).json({ message: 'No ORD items in this order' });
    }
    
    const filename = projectName.replace(/[^a-zA-Z0-9_() -]/g, '_');
    res.setHeader('Content-Type', 'text/plain');
    res.setHeader('Content-Disposition', `attachment; filename="${filename}.ord"`);
    res.send(ord);
  } catch (e: any) {
    res.status(500).json({ message: e.message });
  }
});
```

### Also update the `/data/ord` endpoint to include room grouping

```ts
app.get('/api/orders/:id/data/ord', isAuthenticated, async (req, res) => {
  try {
    const projectId = parseInt(req.params.id);
    const project = await storage.getProject(projectId);
    if (!project) return res.status(404).json({ message: 'Project not found' });
    
    const files = await storage.getProjectFiles(projectId);
    const allItems = await storage.getOrderItemsByProject(projectId);
    
    const rooms = files.map((file, idx) => {
      const fileItems = allItems.filter(i => 
        i.fileId === file.id && i.exportType === 'ORD' && i.exportText
      );
      return {
        roomNumber: idx + 1,
        fileId: file.id,
        fileName: file.originalFilename,
        roomName: file.poNumber || file.originalFilename.replace(/\.[^.]+$/, ''),
        itemCount: fileItems.length,
        items: fileItems.map(i => ({
          sku: i.sku, qty: i.quantity, height: i.height, width: i.width, depth: i.depth,
          exportText: i.exportText,
        })),
      };
    }).filter(r => r.itemCount > 0);
    
    res.json({
      projectName: project.name,
      rooms,
      totalItems: rooms.reduce((s, r) => s + r.itemCount, 0),
    });
  } catch (e: any) { res.status(500).json({ message: e.message }); }
});
```
