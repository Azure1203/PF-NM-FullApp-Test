We need to improve the Cabinet Vision .ORD file generation so that:

Each per-file export starts with the proper [Header] block (not just a stream of [Catalog] blocks)
The ordExporter.ts service is aware of both the file-level header context (PO number, project name) and the item-level context (dimensions, material, banding)
An ord_header_template setting is stored in the database and editable in the UI, so the header format can be adjusted without code changes

Study the reference .ORD files carefully before proceeding. The correct structure of a Cabinet Vision .ORD file is:
[Header]
Version=4
Unit=1
Name="<design name>"
Description="<design name>"
PurchaseOrder="<PO number>"
Comment=""
Customer="Perfect Fit Closets"
Address1="100-111 5 Avenue Southwest"
<item block 1>
<item block 2>
...
Where each item block looks like:
[Catalog]
Name="Perfect Fit"
CabinetConstruction="X-Perfect Fit"
DrawerBoxConstruction="X-Perfect Fit Drawer 19"
RollOutConstruction="X-Perfect Fit Roll Out 19"
Materials="{{color.panel_export}}"

[Parameters]
Attribute="Banding","xPFC_BAND","text","{{edge_left.export}}{{edge_right.export}}{{edge_top.export}}{{edge_bottom.export}}"

[Cabinets]
1,"{{product_name}}",{{width}},{{height}},{{thickness}},"*","N",{{quantity}}
;
The [Header] block appears exactly once per file at the very top. Every item block follows it with no second [Header].

Part 3 — Update ordExporter.ts
Add a new exported function generateOrdHeader:
typescript/**
 * Generates the [Header] block for a Cabinet Vision .ORD file.
 * Called once per file, before all item blocks.
 */
export function generateOrdHeader(
  headerTemplate: string,
  fileContext: {
    designName: string;   // e.g. "H Holtermann (Holtermann V2)"
    poNumber: string;     // e.g. "1919"
  }
): string {
  return headerTemplate
    .replace(/\{\{design_name\}\}/g, fileContext.designName)
    .replace(/\{\{po_number\}\}/g, fileContext.poNumber);
}

In client/src/pages/OrderDetails.tsx, update the Download .ORD button logic to reconstruct the correct per-file structure:
Instead of just joining all exportText values together, group items by fileId and prepend a header for each file:
typescriptconst handleDownloadOrd = async () => {
  if (!orderItems || orderItems.length === 0) {
    toast({ title: "No export data", description: "Run pricing first.", variant: "destructive" });
    return;
  }
  
  // Group items by fileId
  const byFile = new Map<number, typeof orderItems>();
  for (const item of orderItems) {
    if (!item.exportText) continue;
    if (!byFile.has(item.fileId)) byFile.set(item.fileId, []);
    byFile.get(item.fileId)!.push(item);
  }
  
  if (byFile.size === 0) {
    toast({ title: "No export text", description: "No items have export text. Check product export formulas.", variant: "destructive" });
    return;
  }
  
  // Fetch the project's order files to get PO numbers and filenames
  const filesRes = await fetch(`/api/orders/${id}/files`);
  const files: OrderFile[] = await filesRes.json();
  const fileMap = new Map(files.map(f => [f.id, f]));
  
  // Fetch the header template setting
  const settingRes = await fetch('/api/admin/settings/ord_header_template');
  const { value: headerTemplate } = await settingRes.json();
  
  let ordContent = '';
  for (const [fileId, items] of byFile) {
    const file = fileMap.get(fileId);
    const designName = file?.poNumber || file?.originalFilename?.replace('.csv', '') || 'Unknown';
    const poNumber = (file?.poNumber || '').replace(/[^0-9]/g, '') || '0000';
    
    // Build header
    const header = (headerTemplate || '')
      .replace(/\{\{design_name\}\}/g, designName)
      .replace(/\{\{po_number\}\}/g, poNumber);
    
    ordContent += header + '\n';
    for (const item of items) {
      ordContent += item.exportText + '\n';
    }
  }
  
  const blob = new Blob([ordContent], { type: 'text/plain' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `${project?.name || 'order'}.ord`;
  a.click();
  URL.revokeObjectURL(url);
};
