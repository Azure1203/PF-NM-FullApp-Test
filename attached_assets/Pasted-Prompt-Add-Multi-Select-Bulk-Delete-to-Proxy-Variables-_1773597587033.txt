tsconst bulkDeleteMutation = useMutation({
  mutationFn: async (ids: number[]) => {
    await Promise.all(ids.map(id =>
      apiRequest("DELETE", `/api/admin/proxy-variables/${id}`)
    ));
  },
  onSuccess: () => {
    queryClient.invalidateQueries({ queryKey: ["/api/admin/proxy-variables"] });
    setSelectedIds(new Set());
    setSelectMode(false);
    setEditingId(null);
    form.reset({
      name: "",
      type: "pricing",
      formula: "// Enter your formula here\n",
    });
    toast({ title: "Deleted", description: `${selectedIds.size} variable${selectedIds.size !== 1 ? 's' : ''} deleted` });
  },
  onError: (error: Error) => {
    toast({ title: "Error", description: error.message, variant: "destructive" });
  },
});
