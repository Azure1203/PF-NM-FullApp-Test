Part 2 — New product import route in server/routes.ts
Add this single route after the existing allmoxy-products routes:
tsapp.post('/api/admin/allmoxy-products/import-csv', isAuthenticated, upload.single('file'), async (req, res) => {
  try {
    if (!req.file) return res.status(400).json({ message: 'No file uploaded' });

    const { pricingProxyId, exportProxyId, gridId, alias, lookupColumn } = req.body;

    const records: any[] = parseSync(req.file.buffer.toString('utf-8'), {
      columns: true,
      skip_empty_lines: true,
    });

    if (!records.length) {
      return res.status(400).json({ message: 'File is empty' });
    }

    // Validate this is a PF Products file
    if (!records[0]['PRODUCT NAME'] && !records[0]['ID']) {
      return res.status(400).json({ message: 'This does not look like a PF Products file. Expected columns: ID, PRODUCT NAME' });
    }

    // Derive category name from filename
    const filename = req.file.originalname;
    const categoryName = filename
      .replace(/\.csv$/i, '')
      .replace(/_\d{8}$/, '')
      .replace(/^PF_/, '')
      .replace(/_Products$/, '')
      .replace(/_/g, ' ')
      .trim();

    // Build products list
    const productsToInsert = records
      .filter(r => r['PRODUCT NAME']?.trim())
      .map(r => ({
        name: r['PRODUCT NAME'].trim(),
        skuPrefix: r['PRODUCT NAME'].trim(),
        status: 'active' as const,
        pricingProxyId: pricingProxyId ? Number(pricingProxyId) : null,
        exportProxyId: exportProxyId ? Number(exportProxyId) : null,
        description: categoryName,
        notes: null,
        imageBase64: null,
      }));

    if (!productsToInsert.length) {
      return res.status(400).json({ message: 'No valid product rows found in file' });
    }

    // Wipe existing products in this category and insert fresh
    await storage.deleteAllmoxyProductsByCategory(categoryName);
    const inserted = await storage.bulkInsertAllmoxyProducts(productsToInsert);

    // Apply grid binding to all inserted products if a grid was selected
    let bindingsCreated = 0;
    if (gridId) {
      const bindingAlias = alias?.trim() || categoryName.toLowerCase().replace(/[^a-z0-9]/g, '_');
      const bindingLookupColumn = lookupColumn?.trim() || 'MANU_CODE';
      for (const product of inserted) {
        await storage.replaceProductGridBindings(product.id, [{
          productId: product.id,
          gridId: Number(gridId),
          alias: bindingAlias,
          lookupColumn: bindingLookupColumn,
        }]);
        bindingsCreated++;
      }
    }

    res.json({
      categoryName,
      productsInserted: inserted.length,
      bindingsCreated,
      gridId: gridId ? Number(gridId) : null,
      pricingProxyId: pricingProxyId ? Number(pricingProxyId) : null,
      exportProxyId: exportProxyId ? Number(exportProxyId) : null,
    });
  } catch (e: any) {
    res.status(500).json({ message: e.message });
  }
});

Add the import handler:
tsconst handleImport = async () => {
  if (!importFile) return;
  setImportLoading(true);
  setImportResult(null);
  try {
    const formData = new FormData();
    formData.append('file', importFile);
    if (importPricingProxyId) formData.append('pricingProxyId', String(importPricingProxyId));
    if (importExportProxyId) formData.append('exportProxyId', String(importExportProxyId));
    if (importGridId) formData.append('gridId', String(importGridId));
    if (importAlias) formData.append('alias', importAlias);
    formData.append('lookupColumn', importLookupColumn || 'MANU_CODE');

    const res = await fetch('/api/admin/allmoxy-products/import-csv', {
      method: 'POST',
      body: formData,
    });
    const data = await res.json();
    if (!res.ok) throw new Error(data.message);

    setImportResult(data);
    queryClient.invalidateQueries({ queryKey: ['/api/admin/allmoxy-products'] });
    toast({
      title: 'Import complete',
      description: `${data.productsInserted} products imported for ${data.categoryName}`,
    });
  } catch (e: any) {
    toast({ title: 'Import failed', description: e.message, variant: 'destructive' });
  } finally {
    setImportLoading(false);
  }
};
