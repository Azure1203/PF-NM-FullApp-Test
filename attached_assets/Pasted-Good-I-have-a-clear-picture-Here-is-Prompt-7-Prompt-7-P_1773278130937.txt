Step 6 — Add a POST /api/orders/:id/regenerate-checklists endpoint

This is a convenience endpoint to regenerate all checklists for a project without re-uploading the CSV. Add it:
typescript

app.post('/api/orders/:id/regenerate-checklists', isAuthenticated, async (req, res) => {
  try {
    const projectId = parseInt(req.params.id);
    const files = await storage.getProjectFiles(projectId);
    
    let totalHardwareItems = 0;
    let totalPackingItems = 0;
    let errors: string[] = [];
    
    for (const file of files) {
      if (!file.rawContent) continue;
      
      const hwResult = await generateHardwareChecklistForFile(file.id, file.rawContent);
      const psResult = await generatePackingSlipChecklistForFile(file.id, file.rawContent);
      
      if (hwResult.success) totalHardwareItems += hwResult.itemCount;
      else errors.push(`File ${file.id} hardware: ${hwResult.error}`);
      
      if (psResult.success) totalPackingItems += psResult.itemCount;
      else errors.push(`File ${file.id} packing slip: ${psResult.error}`);
    }
    
    res.json({ 
      success: errors.length === 0, 
      totalHardwareItems, 
      totalPackingItems,
      errors 
    });
  } catch (e: any) {
    res.status(500).json({ message: e.message });
  }
});
